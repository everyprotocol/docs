# Get started

## Accounts

* cli: --network --universe matter/kind/value/wallet/balance
* cli: --foundry account not recognized
every wallet inspect alice --foundry
Cannot read properties of undefined (reading 'content')

```bash
NAMES=(alice bob charlie dave)
MNEMOIC="test test test test test test test test test test test junk"
PASSWORD=123
N=${#NAMES}
# N=3
for i in {1..$N}; do
    index=$(( i-1 ))
    name="${NAMES[$i]}"
    cast wallet import $name --unsafe-password $PASSWORD \
        --mnemonic "$MNEMOIC" --mnemonic-index $index
done
```
```bash
NAMES=(alice bob charlie dave)
PASSWORD=123
N=${#NAMES}
# N=3
for i in {1..$N}; do
    index=$(( i-1 ))
    name="${NAMES[$i]}"
    suri="//$(tr '[:lower:]' '[:upper:]' <<< ${name:0:1})${name:1}"
    every wallet import $name -p $PASSWORD -t sr25519 $suri
done
```

```bash

    --foundry --account charlie --observer localnet
```

## Create Sets

```bash
forge init -t everytemplate/set dice-set
cd dice-set
forge build
```

```bash
forge create src/DiceSet.sol:DiceSet \
    --rpc-url http://anvil.localhost \
    --account charlie \
    --constructor-args 0x854C35Fd2b65fE9fcE71dddE91De8c3e1A7Dc8Ae 17 1
```

```bash
forge verify-contract \
    --rpc-url http://anvil.localhost \
    --verifier blockscout \
    --verifier-url http://scout.localhost/api/ \
    0x0DCd1Bf9A1b36cE34237eEaFef220932846BCD82 \
    src/DiceSet.sol:DiceSet
```
